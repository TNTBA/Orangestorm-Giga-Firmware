148  /// Get the remote endpoint address
149  /**
150  * The iostream transport has no information about the ultimate remote
151  * endpoint. It will return the string "iostream transport". To indicate
152  * this.
153  *
154  * TODO: allow user settable remote endpoint addresses if this seems useful
155  *
156  * @return A string identifying the address of the remote endpoint
157  */
158  std::string get_remote_endpoint(lib::error_code & ec) const {
159  std::stringstream s;
160 
161  lib::asio::error_code aec;
162  lib::asio::ip::tcp::endpoint ep = m_socket->lowest_layer().remote_endpoint(aec);
163 
164  if (aec) {
165  ec = error::make_error_code(error::pass_through);
166  s << "Error getting remote endpoint: " << aec
167  << " (" << aec.message() << ")";
168  return s.str();
169  } else {
170  ec = lib::error_code();
171  s << ep;
172  return s.str();
173  }

226  /// Pre-initialize security policy
227  /**
228  * Called by the transport after a new connection is created to initialize
229  * the socket component of the connection. This method is not allowed to
230  * write any bytes to the wire. This initialization happens before any
231  * proxies or other intermediate wrappers are negotiated.
232  *
233  * @param callback Handler to call back with completion information
234  */
235  void pre_init(init_handler callback) {
236  // TODO: is this the best way to check whether this function is
237  // available in the version of OpenSSL being used?
238  // TODO: consider case where host is an IP address
239 #if OPENSSL_VERSION_NUMBER >= 0x90812f
240  if (!m_is_server) {
241  // For clients on systems with a suitable OpenSSL version, set the
242  // TLS SNI hostname header so connecting to TLS servers using SNI
243  // will work.
244  long res = SSL_set_tlsext_host_name(
245  get_socket().native_handle(), m_uri->get_host().c_str());
246  if (!(1 == res)) {
247  callback(socket::make_error_code(socket::error::tls_failed_sni_hostname));
248  }
249  }
250 #endif
251 
252  callback(lib::error_code());
253  }

255  /// Post-initialize security policy
256  /**
257  * Called by the transport after all intermediate proxies have been
258  * negotiated. This gives the security policy the chance to talk with the
259  * real remote endpoint for a bit before the websocket handshake.
260  *
261  * @param callback Handler to call back with completion information
262  */
263  void post_init(init_handler callback) {
264  m_ec = socket::make_error_code(socket::error::tls_handshake_timeout);
265 
266  // TLS handshake
267  if (m_strand) {
268  m_socket->async_handshake(
269  get_handshake_type(),
270  m_strand->wrap(lib::bind(
271  &type::handle_init, get_shared(),
272  callback,
273  lib::placeholders::_1
274  ))
275  );
276  } else {
277  m_socket->async_handshake(
278  get_handshake_type(),
279  lib::bind(
280  &type::handle_init, get_shared(),
281  callback,
282  lib::placeholders::_1
283  )
284  );
285  }
286  }

337  /// Translate any security policy specific information about an error code
338  /**
339  * Translate_ec takes an Asio error code and attempts to convert its value
340  * to an appropriate websocketpp error code. In the case that the Asio and
341  * Websocketpp error types are the same (such as using boost::asio and
342  * boost::system_error or using standalone asio and std::system_error the
343  * code will be passed through natively.
344  *
345  * In the case of a mismatch (boost::asio with std::system_error) a
346  * translated code will be returned. Any error that is determined to be
347  * related to TLS but does not have a more specific websocketpp error code
348  * is returned under the catch all error `tls_error`. Non-TLS related errors
349  * are returned as the transport generic error `pass_through`
350  *
351  * @since 0.3.0
352  *
353  * @param ec The error code to translate_ec
354  * @return The translated error code
355  */
356  template <typename ErrorCodeType>
357  static
358  lib::error_code translate_ec(ErrorCodeType ec) {
359  if (ec.category() == lib::asio::error::get_ssl_category()) {
360  // We know it is a TLS related error, but otherwise don't know more.
361  // Pass through as TLS generic.
362  return make_error_code(transport::error::tls_error);
363  } else {
364  // We don't know any more information about this error so pass
365  // through
366  return make_error_code(transport::error::pass_through);
367  }
368  }
369 
370  static
371  /// Overload of translate_ec to catch cases where lib::error_code is the
372  /// same type as lib::asio::error_code
373  lib::error_code translate_ec(lib::error_code ec) {
374  return ec;
375  }
