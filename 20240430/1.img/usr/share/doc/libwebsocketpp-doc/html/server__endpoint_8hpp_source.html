96  /// Create and initialize a new connection
97  /**
98  * The connection will be initialized and ready to begin. Call its start()
99  * method to begin the processing loop.
100  *
101  * Note: The connection must either be started or terminated using
102  * connection::terminate in order to avoid memory leaks.
103  *
104  * @return A pointer to the new connection.
105  */
106  connection_ptr get_connection() {
107  return endpoint_type::create_connection();
108  }
109 
110  /// Starts the server's async connection acceptance loop (exception free)
111  /**
112  * Initiates the server connection acceptance loop. Must be called after
113  * listen. This method will have no effect until the underlying io_service
114  * starts running. It may be called after the io_service is already running.
115  *
116  * Refer to documentation for the transport policy you are using for
117  * instructions on how to stop this acceptance loop.
118  *
119  * @param [out] ec A status code indicating an error, if any.
120  */
121  void start_accept(lib::error_code & ec) {
122  if (!transport_type::is_listening()) {
123  ec = error::make_error_code(error::async_accept_not_listening);
124  return;
125  }
126 
127  ec = lib::error_code();
128  connection_ptr con = get_connection();
129 
130  if (!con) {
131  ec = error::make_error_code(error::con_creation_failed);
132  return;
133  }
134 
135  transport_type::async_accept(
136  lib::static_pointer_cast<transport_con_type>(con),
137  lib::bind(&type::handle_accept,this,con,lib::placeholders::_1),
138  ec
139  );
140 
141  if (ec && con) {
142  // If the connection was constructed but the accept failed,
143  // terminate the connection to prevent memory leaks
144  con->terminate(lib::error_code());
145  }
146  }
147 
148  /// Starts the server's async connection acceptance loop
149  /**
150  * Initiates the server connection acceptance loop. Must be called after
151  * listen. This method will have no effect until the underlying io_service
152  * starts running. It may be called after the io_service is already running.
153  *
154  * Refer to documentation for the transport policy you are using for
155  * instructions on how to stop this acceptance loop.
156  */
157  void start_accept() {
158  lib::error_code ec;
159  start_accept(ec);
160  if (ec) {
161  throw exception(ec);
162  }
163  }
164 
165  /// Handler callback for start_accept
166  void handle_accept(connection_ptr con, lib::error_code const & ec) {
167  if (ec) {
168  con->terminate(ec);
169 
170  if (ec == error::operation_canceled) {
171  endpoint_type::m_elog->write(log::elevel::info,
172  "handle_accept error: "+ec.message());
173  } else {
174  endpoint_type::m_elog->write(log::elevel::rerror,
175  "handle_accept error: "+ec.message());
176  }
177  } else {
178  con->start();
179  }
180 
181  lib::error_code start_ec;
182  start_accept(start_ec);
183  if (start_ec == error::async_accept_not_listening) {
184  endpoint_type::m_elog->write(log::elevel::info,
185  "Stopping acceptance of new connections because the underlying transport is no longer listening.");
186  } else if (start_ec) {
187  endpoint_type::m_elog->write(log::elevel::rerror,
188  "Restarting async_accept loop failed: "+ec.message());
189  }
190  }
191 };
